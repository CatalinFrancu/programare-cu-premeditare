\begin{figure}[H]
  \centering

  \begin{tikzpicture}[scale=0.8, every node/.style={scale=0.8}]
    \coordinate (O) at (0,0);
    \coordinate (X) at (14,0);
    \coordinate (A) at (1,4);
    \coordinate (AB) at (13,0);
    \coordinate (B) at ($(A)!0.1!(AB)$); % 10% from A and 90% from AB
    \coordinate (BC) at (11.0,0);
    \coordinate (C) at ($(B)!0.2!(BC)$);
    \coordinate (CD) at (9,0);
    \coordinate (D) at ($(C)!0.2!(CD)$);
    \coordinate (DE) at (7.5,0);
    \coordinate (E) at ($(D)!0.2!(DE)$);
    \coordinate (EF) at (6.7,0);
    \coordinate (F) at ($(E)!0.4!(EF)$);
    \coordinate (P) at (7.9, 1.5);
    \coordinate (XP) at ($(O)!(P)!(X)$); % projection of P on the OX line

    \fill[black!10] (B) -- (AB) -- (BC) -- cycle;

    \fill (A) circle(0.05) node[below left] {$A$};
    \fill (B) circle(0.05) node[below left] {$B$};
    \fill (C) circle(0.05) node[below left] {$C$};
    \fill (D) circle(0.05) node[below left] {$D$};
    \fill (E) circle(0.05) node[below left] {$E$};
    \fill (F) circle(0.05) node[below left] {$F$};
    \fill (P) circle(0.05) node[below right] {$P$};
    \fill (AB) circle(0.05) node[below] {$x_{AB}$};
    \fill (BC) circle(0.05) node[below] {$x_{BC}$};
    \fill (CD) circle(0.05) node[below] {$x_{CD}$};
    \fill (DE) circle(0.05) node[below] {$x_{DE}$};
    \fill (EF) circle(0.05) node[below] {$x_{EF}$};
    \fill (XP) circle(0.05) node[below right] {$x_P$};

    \draw[-{Stealth}] (O) -- (X);
    \draw (A) -- ($(O)!(A)!(X)$);
    \draw (A) -- (B);
    \draw[dashed] (B) -- (AB);
    \draw (B) -- ($(O)!(B)!(X)$);
    \draw (B) -- (C);
    \draw[dashed] (C) -- (BC);
    \draw (C) -- ($(O)!(C)!(X)$);
    \draw (C) -- (D);
    \draw[dashed] (D) -- (CD);
    \draw (D) -- ($(O)!(D)!(X)$);
    \draw (D) -- (E);
    \draw[dashed] (E) -- (DE);
    \draw (E) -- ($(O)!(E)!(X)$);
    \draw (E) -- (F);
    \draw[dashed] (F) -- (EF);
    \draw (F) -- ($(O)!(F)!(X)$);
    \draw (P) -- (XP);
  \end{tikzpicture}

  \caption{Mentorul vrăjitorului $P$ este $D$ deoarece $x_P$ se află în triunghiul lui $D$. Mai mult, punctul $P$ se află în triunghiul lui $B$, deci vrăjitorii $C-F$ nu vor mai fi mentori niciodată.}

\end{figure}
